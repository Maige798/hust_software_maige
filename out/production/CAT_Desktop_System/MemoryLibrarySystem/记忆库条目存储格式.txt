// 加了斜杠是代表注释（实际文件中不存在任何注释）
// 文件后缀为 .mlib
// 任意两个条目中间存在一行全空

// 此条目存储文件名称（必须是文件的第一个条目）
@Name {
	<name> 文件名称
}

// 此条目存储文件路径（必须是文件的第二个条目）
@Save {
	<save> 文件路径
}

// 此条目存储原文与译文都有内容的条目
@TranItem {
	<ori> [原文语言] 原文内容
	<tran> [译文语言] 译文内容
}

// 此条目存储只有原文有内容的条目（记忆库原则上不允许出现，但项目库允许，而项目库格式与记忆库基本一致，故保留此类条目）
@TranItem {
	<ori> [原文语言] 原文内容
	<blank>
}

// 此条目存储条目空格（" "）
@Blank

// 此条目存储条目制表符（"\t"）
@Table

// 此条目存储条目回车符与换行符（"\r\n"）
@Return
